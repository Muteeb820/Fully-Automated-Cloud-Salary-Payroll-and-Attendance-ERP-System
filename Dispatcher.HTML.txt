  <script>
    function createParticles() {
        const container = document.getElementById('particles');
        container.style.display = 'block';
        for(let i=0; i<20; i++) {
            let particle = document.createElement('div');
            particle.className = 'particle';
            particle.style.left = Math.random() * 100 + '%';
            particle.style.animationDuration = (Math.random() * 1 + 0.5) + 's';
            particle.style.animationDelay = Math.random() * 2 + 's';
            container.appendChild(particle);
        }
    }

    function runGoogleScript(func, ...args) {
        return new Promise((resolve, reject) => {
            google.script.run.withSuccessHandler(resolve).withFailureHandler(reject)[func](...args);
        });
    }

    function updateTerminal(msg) {
        const term = document.getElementById('terminalLog');
        term.innerHTML += `<br>> ${msg}`;
        term.scrollTop = term.scrollHeight; 
    }

    async function startBulkProcess() {
        const month = document.getElementById('targetMonth').value.trim();
        if(!month) {
            alert("Authorization Failed: Valid month parameter is required.");
            return;
        }

        const uiHeader = document.getElementById('uiHeader');
        const pCont = document.getElementById('progressContainer');
        const pBar = document.getElementById('progressBar');
        const statusMsg = document.getElementById('statusMessage');
        const countTxt = document.getElementById('countText');
        const term = document.getElementById('terminalLog');

        uiHeader.style.display = "none";
        pCont.style.display = "block";
        term.innerHTML = `> Initiating sequence for parameter: ${month}...`;
        createParticles();

        try {
            updateTerminal("Querying primary database...");
            const allEmployees = await runGoogleScript('getEmployeeDataForBatching', month);
            const total = allEmployees.length;
            
            if(total === 0) {
                statusMsg.innerText = "Error: Protocol Aborted"; 
                statusMsg.style.color = "var(--danger)";
                updateTerminal("<span style='color:#ef4444'>FATAL: No valid active records found for the specified parameter.</span>");
                document.getElementById('particles').style.display = 'none';
                return;
            }

            updateTerminal(`Identified ${total} valid records. Initializing Batch Processor...`);
            let processedCount = 0;
            const batchSize = 5; // Set batch size for operational stability..........................................................................................................................................................------------------------------------

            for (let i = 0; i < total; i += batchSize) {
                const batch = allEmployees.slice(i, i + batchSize);
                statusMsg.innerText = `Processing Batch ${Math.floor(i/batchSize) + 1}...`;
                updateTerminal(`Rendering document payload for records ${i+1} to ${Math.min(i+batchSize, total)}...`);
                
                const successCount = await runGoogleScript('processSingleBatch', month, batch);
                processedCount += successCount;
                
                let percent = Math.round((processedCount / total) * 100);
                pBar.style.width = percent + '%';
                countTxt.innerText = `${processedCount} / ${total}`;
                updateTerminal(`Batch processed successfully. Enforcing anti-spam protocol delay.`);
            }

            updateTerminal("<span style='color:#10b981'>All operations executed successfully.</span>");
            document.getElementById('particles').style.display = 'none';
            
            setTimeout(() => {
                pCont.style.display = "none";
                document.getElementById('mainCard').style.borderColor = "var(--success)";
                document.getElementById('successContainer').style.display = "block";
            }, 1000);

        } catch (error) {
            statusMsg.innerText = "System Exception!"; 
            statusMsg.style.color = "var(--danger)";
            updateTerminal(`<span style='color:#ef4444'>ERROR: ${error.message}</span>`);
            document.getElementById('particles').style.display = 'none';
        }
    }
  </script>
