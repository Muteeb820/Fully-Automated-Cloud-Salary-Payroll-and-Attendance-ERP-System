  <script>
    let chart;
    
    // FBR Tax Slabs
    function getTax(sal) {
      if (sal <= 50000) return 0; if (sal <= 60000) return 100; if (sal <= 70000) return 200;
      if (sal <= 80000) return 300; if (sal <= 90000) return 400; if (sal <= 100000) return 500;
      if (sal <= 110000) return 1600; if (sal <= 120000) return 2700; if (sal <= 130000) return 3800;
      if (sal <= 140000) return 4900; if (sal <= 150000) return 6000; if (sal <= 160000) return 7100;
      if (sal <= 170000) return 8200; if (sal <= 180000) return 9300; if (sal <= 190000) return 11200;
      return 13500;
    }

    function initChart() {
      const ctx = document.getElementById('deductionChart').getContext('2d');
      chart = new Chart(ctx, {
        type: 'doughnut',
        data: {
          labels: ['Net Salary', 'Deductions'],
          datasets: [{ data: [100, 0], backgroundColor: ['#10b981', '#ef4444'], borderWidth: 0, cutout: '75%' }]
        },
        options: { plugins: { legend: { display: false } }, responsive: true, maintainAspectRatio: false }
      });
    }

    function calculate() {
      const sal = parseFloat(document.getElementById('salary').value) || 0;
      const late = parseFloat(document.getElementById('late').value) || 0;
      const bonus = parseFloat(document.getElementById('bonus').value) || 0;
      const unpaid = parseFloat(document.getElementById('unpaid').value) || 0;
      const sandwich = parseFloat(document.getElementById('sandwich').value) || 0;
      const fine = parseFloat(document.getElementById('fine').value) || 0;
      
      const medTier = parseInt(document.getElementById('medTier').value) || 0;
      const remLoan = parseFloat(document.getElementById('remLoan').value) || 0;
      const loanInst = parseFloat(document.getElementById('loanInst').value) || 0;
      
      const dayRate = sal / 30; 
      const minRate = sal / 16200;
      
      const lateDed = minRate * late; 
      const absPen = dayRate * (unpaid + sandwich);
      const tax = getTax(sal); 
      const pfFixed = 680;
      const medDed = medTier * 600; 
      
      let actualLoanDeduction = 0;
      if (remLoan > 0) {
        actualLoanDeduction = loanInst;
        if (actualLoanDeduction > remLoan) {
            actualLoanDeduction = remLoan; 
        }
      }
      
      const totalDed = lateDed + absPen + fine + tax + pfFixed + medDed + actualLoanDeduction;
      const grossEarn = sal + bonus;
      const net = grossEarn - totalDed;

      const fmt = (num) => Math.round(num).toLocaleString();

      document.getElementById('rateDay').innerText = "PKR " + fmt(dayRate);
      document.getElementById('rateMin').innerText = "PKR " + minRate.toFixed(2);
      
      document.getElementById('vLate').innerText = "- PKR " + fmt(lateDed);
      document.getElementById('vAbsPen').innerText = "- PKR " + fmt(absPen);
      document.getElementById('vTaxFixed').innerText = "- PKR " + fmt(tax + pfFixed);
      document.getElementById('vMedLoan').innerText = "- PKR " + fmt(medDed + actualLoanDeduction);
      document.getElementById('netVal').innerText = "PKR " + fmt(net > 0 ? net : 0);
      
      const totalPie = grossEarn > 0 ? grossEarn : 1;
      document.getElementById('pTake').innerText = Math.round((Math.max(0, net)/totalPie)*100) + '%';
      document.getElementById('pDed').innerText = Math.round((Math.min(grossEarn, totalDed)/totalPie)*100) + '%';
      chart.data.datasets[0].data = [net > 0 ? net : 0, totalDed];
      chart.update();

      document.getElementById('s-basic').innerText = fmt(sal);
      document.getElementById('s-bonus').innerText = "+ " + fmt(bonus);
      document.getElementById('s-late').innerText = "- " + fmt(lateDed);
      document.getElementById('s-absent').innerText = "- " + fmt(absPen);
      document.getElementById('s-fine').innerText = "- " + fmt(fine);
      document.getElementById('s-tax').innerText = "- " + fmt(tax);
      document.getElementById('s-med').innerText = "- " + fmt(medDed);
